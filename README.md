class LiveClient {
  constructor(serverUrl) {
    this.ws = new WebSocket(serverUrl);
  }

  joinRoom(roomId, user) {
    this.ws.send(JSON.stringify({
      type: "join",
      roomId,
      user
    }));
  }

  sendMessage(message) {
    this.ws.send(JSON.stringify({
      type: "chat",
      message,
      timestamp: Date.now()
    }));
  }
}
